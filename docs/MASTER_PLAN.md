#### 4. **Error Handling**
**Target:** Graceful degradation, user-friendly errors

**Before:**
```typescript
const result = await supabase.from('analyses').insert(data);
// No error handling - crashes plugin
```

**After:**
```typescript
try {
  const result = await supabase.from('analyses').insert(data);
  if (result.error) throw result.error;
  
  showToast('Analysis saved!', 'success');
} catch (error) {
  console.error('Failed to save analysis:', error);
  
  // Fallback: Save to local cache only
  saveToLocalCache(data);
  
  showToast(
    'Could not sync to cloud. Saved locally instead.',
    'warning'
  );
}
```

import { debounce } from './utils/debounce';
